def marshal_load array
  @name       = array[1]
  @full_name  = array[2]
  @rw         = array[3]
  @visibility = array[4]
  @comment    = array[5]

  @parent_name = @full_name
end

def type

  when 'RW' then 'attr_accessor'
  when 'R'  then 'attr_reader'
  when 'W'  then 'attr_writer'
  end
